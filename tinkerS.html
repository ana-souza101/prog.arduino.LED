void setup()
{
  pinMode(A2, OUTPUT);
  pinMode(4, OUTPUT);
  pinMode(A1, OUTPUT);
  pinMode(7, OUTPUT);
  pinMode(A0, OUTPUT);
  pinMode(8, OUTPUT);
}

void loop()
{
  digitalWrite(A2, HIGH);
  digitalWrite(4, HIGH);
  delay(5000); // Wait for 5000 millisecond(s)
  digitalWrite(A2, LOW);
  digitalWrite(4, LOW);
  digitalWrite(A1, HIGH);
  digitalWrite(7, HIGH);
  delay(3000); // Wait for 3000 millisecond(s)
  digitalWrite(A1, LOW);
  digitalWrite(7, LOW);
  digitalWrite(A0, HIGH);
  digitalWrite(8, HIGH);
  delay(5000); // Wait for 5000 millisecond(s)
  digitalWrite(A0, LOW);
  digitalWrite(8, LOW);
  delay(1); // Wait for 1 millisecond(s)
}
